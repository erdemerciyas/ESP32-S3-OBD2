static uint16_t s_gatt_ready_conn;

static int notify_subscribe_cb(uint16_t conn_handle, const struct ble_gatt_error *error,
                               struct ble_gatt_attr *attr, void *arg)
{
    (void)attr;
    (void)arg;
    if (error->status != 0) {
        app_log_error(TAG, "Notify subscribe failed: %d", error->status);
        ble_gap_terminate(conn_handle, BLE_ERR_REM_USER_CONN_TERM);
        return 0;
    }
    app_log_info(TAG, "Notifications enabled");
    vTaskDelay(pdMS_TO_TICKS(350));
    on_gatt_ready(s_gatt_ready_conn);
    return 0;
}
