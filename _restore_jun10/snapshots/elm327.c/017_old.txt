static bool send_raw(const char *cmd);
static bool wait_response(uint32_t timeout_ms);
static void drain_sync_sem(void);

static bool init_cmd_critical(const char *cmd)
{
    return strcmp(cmd, "ATZ") == 0;
}

static uint32_t init_cmd_timeout_ms(const char *cmd, const char *protocol_cmd)
{
    if (strcmp(cmd, "ATZ") == 0) {
        return 5000;
    }
    if (strcmp(cmd, protocol_cmd) == 0) {
        return 12000;
    }
    return 2500;
}

static bool send_init_cmd(const char *cmd, uint32_t timeout_ms, uint32_t pre_delay_ms, int max_attempts)
{
    for (int attempt = 0; attempt < max_attempts; attempt++) {
        if (!send_raw(cmd)) {
            return false;
        }
        if (pre_delay_ms > 0) {
            vTaskDelay(pdMS_TO_TICKS(pre_delay_ms));
        }
        if (wait_response(timeout_ms)) {
            return true;
        }
        ESP_LOGW(TAG, "Timeout on %s (attempt %d/%d)", cmd, attempt + 1, max_attempts);
        drain_sync_sem();
    }
    return false;
}
