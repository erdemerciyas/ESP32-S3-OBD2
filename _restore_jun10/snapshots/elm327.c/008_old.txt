static bool init_cmd_critical(const char *cmd, const char *protocol_cmd)
{
    return strcmp(cmd, "ATZ") == 0 || strcmp(cmd, protocol_cmd) == 0;
}

static uint32_t init_cmd_timeout_ms(const char *cmd, const char *protocol_cmd)
{
    if (strcmp(cmd, "ATZ") == 0) {
        return 6000;
    }
    if (strcmp(cmd, protocol_cmd) == 0) {
        return 12000;
    }
    return 3000;
}
